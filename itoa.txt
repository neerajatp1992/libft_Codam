#include <stdlib.h>

static int	ft_len(int n)
{
	int len = 0;

	if (n <= 0)
		len++; // for '-' or '0'
	while (n)
	{
		n /= 10;
		len++;
	}
	return len;
}

char	*ft_itoa(int n)
{
	char	*res;
	int		len;
	long	nb;

	nb = n;
	len = ft_len(n);

	res = (char *)malloc(sizeof(char) * (len + 1));
	if (!res)
		return NULL;

	res[len] = '\0';

	if (nb == 0)
		res[0] = '0';

	if (nb < 0)
	{
		res[0] = '-';
		nb = -nb;
	}

	while (nb > 0)
	{
		res[--len] = (nb % 10) + '0';
		nb /= 10;
	}

	return res;
}
